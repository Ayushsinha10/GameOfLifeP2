private static InputStream fetch(URL url, String urlencode) {
        try {
        String urlEncoded = URLEncoder.encode(urlencode, "UTF-8");
        String path = "./cache" + "/" + urlEncoded;
        if (!Files.isRegularFile(Paths.get(path))){
            Files.copy(url.openStream(), Paths.get(path));
           

        }
        return new FileInputStream(path);
        }
        catch(IOException e){
            e.printStackTrace();
            return null;
        }
    }
    private static Document fetchXML(URL url, String urlencode){
        try{
        DocumentBuilder db = DocumentBuilderFactory.newInstance().newDocumentBuilder();
        Document doc = db.parse(fetch(url, urlencode));
        doc.getDocumentElement().normalize();
        return doc;
        }
        catch(Exception e){
            e.printStackTrace();
            return null;
        }
    }
